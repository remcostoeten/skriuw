# API Reference

Complete documentation of available hooks, utilities, and APIs in the Notys application.

## Core Hooks

These are the building blocks for all data operations in the app. Every module uses these patterns.

### createQueryHook

Creates a typed query hook with automatic error handling and loading states. This is how ALL queries are built.

```typescript
function createQueryHook<TArgs extends AnyFn, TData = any>(
  buildQuery: TArgs,
  options?: CreateQueryOptions<TData>
): (...args: Parameters<TArgs>) => QueryResult<TData>
```

**Options:**
- `select?: (raw: any) => TData` - Transform raw data
- `initialData?: TData` - Fallback value
- `enabled?: (...args: any[]) => boolean` - Conditional execution
- `showErrorToast?: boolean` - Show error notifications (default: `true`)
- `errorContext?: string` - Error message context
- `onError?: (error: Error) => void` - Error callback

**Returns:**
```typescript
{
  data: TData,
  isLoading: boolean,
  error: Error | null,
  isSuccess: boolean,
  isError: boolean
}
```

**Example:**
```typescript
// modules/notes/api/queries/get-notes.ts
import { createQueryHook } from '@/hooks/core';
import type { Note } from '@/api/db/schema';

const useNotesQuery = createQueryHook(
  () => ({
    notes: {
      $: { order: { createdAt: 'desc' } },
      folder: {},  // Include relations
      tasks: {},
    },
  }),
  {
    select: (raw) => (raw?.notes as Note[]) ?? [],
    initialData: [] as Note[],
    showErrorToast: false,
  }
);

export function useGetNotes() {
  const { data, isLoading, error } = useNotesQuery();
  return { notes: data, isLoading, error };
}
```

**With Parameters:**
```typescript
const useNoteQuery = createQueryHook(
  (noteId: string) => ({
    notes: {
      $: { where: { id: noteId } },
    },
  }),
  {
    select: (raw) => raw?.notes?.[0] ?? null,
    enabled: (noteId) => !!noteId,  // Only query if ID exists
  }
);

// Usage
const { data: note } = useNoteQuery(noteId);
```

### useMutation

Wraps async mutations with loading/error states and callbacks.

```typescript
function useMutation<TData, TVariables = void>(
  mutationFn: (variables: TVariables) => Promise<TData>,
  options?: UseMutationOptions<TData, TVariables>
): MutationResult<TData, TVariables>
```

**Options:**
- `onSuccess?: (data: TData, variables: TVariables) => void | Promise<void>`
- `onError?: (error: Error, variables: TVariables) => void | Promise<void>`
- `showErrorToast?: boolean` - Show error toast (default: `true`)
- `errorContext?: string` - Context for error messages

**Returns:**
```typescript
{
  mutate: (variables: TVariables) => Promise<TData>,
  isLoading: boolean,
  error: Error | null,
  reset: () => void
}
```

**Example - Simple Mutation:**
```typescript
// modules/notes/api/mutations/create.ts
import { useCreate, useMutation } from '@/hooks/core';
import { generateId } from 'utils';

type CreateNoteInput = {
  title: string;
  content: string;
  position: number;
};

export function useCreateNote() {
  const { create } = useCreate('notes');
  
  const { mutate, isLoading, error } = useMutation(
    async (input: CreateNoteInput) => {
      const id = generateId();
      const now = Date.now();
      
      await create(id, {
        ...input,
        createdAt: now,
        updatedAt: now,
      });
      
      return { id, ...input };
    },
    {
      errorContext: 'Creating note',
      onSuccess: (data) => {
        console.log('Note created:', data.id);
      },
    }
  );

  return { createNote: mutate, isLoading, error };
}
```

**Example - Complex Mutation:**
```typescript
export function useUpdateNote() {
  const { update } = useUpdate('notes');
  
  const { mutate, isLoading } = useMutation(
    async ({ id, ...data }: UpdateInput) => {
      await update(id, {
        ...data,
        updatedAt: Date.now(),
      });
    },
    {
      errorContext: 'Updating note',
      showErrorToast: true,
    }
  );

  return { updateNote: mutate, isLoading };
}
```

### useCreate

Entity creation helper with automatic ID generation and timestamps.

```typescript
function useCreate<T extends Record<string, any>>(
  entityName: string
): {
  create: (id: string, data: T) => Promise<void>
}
```

**Example:**
```typescript
const { create } = useCreate('notes');

await create(generateId(), {
  title: 'New Note',
  content: '',
  position: 0,
  createdAt: Date.now(),
  updatedAt: Date.now(),
});
```

## Tauri APIs

### Store

Persistent key-value storage.

```typescript
import { Store } from '@tauri-apps/plugin-store';

const store = new Store('settings.dat');

// Set value
await store.set('key', 'value');

// Get value
const value = await store.get('key');

// Delete value
await store.delete('key');

// Clear all
await store.clear();

// Get all entries
const entries = await store.entries();
```

### Dialog

Native file/folder dialogs.

```typescript
import { open, save } from '@tauri-apps/plugin-dialog';

// Open file picker
const selected = await open({
  multiple: false,

// Save file dialog
```

### File System

File operations.

```typescript
import {
  readTextFile,
  writeTextFile,
  exists,
  remove,
} from '@tauri-apps/plugin-fs';

// Read file
const content = await readTextFile('path/to/file.txt');

// Write file
await writeTextFile('path/to/file.txt', 'content');

// Check existence
```
