## Overview

Implement secure note sharing that allows users to generate public links for individual notes **only in web view**. Desktop app does not expose sharing functionality.

### Core Requirements
1. ✅ Notes private by default
2. ✅ Web-only sharing UI
3. ✅ Isolated public view (no user data exposed)
4. ✅ Revocable links
5. ✅ No authentication required for viewing

## API Hooks

### 1. Share Note

```typescript
// modules/notes/api/mutations/share-note.ts
import { useMutation } from '@/hooks/core';
import { transact, tx } from '@/lib/db/client';

function generateShareSlug(): string {
  const chars = 'ABCDEFGHIJKLMNOPQRSTUVWXYZabcdefghijklmnopqrstuvwxyz0123456789';
  return Array.from({ length: 10 }, () => 
    chars[Math.floor(Math.random() * chars.length)]
  ).join('');
}

export function useShareNote() {
  const { mutate, isLoading, error } = useMutation(
    async (noteId: string) => {
      const shareSlug = generateShareSlug();
      const now = Date.now();

      await transact([
        tx.notes[noteId].update({
          isPublic: true,
          shareSlug,
          sharedAt: now,
        }),
      ]);

      return {
        shareUrl: `${window.location.origin}/share/${shareSlug}`,
        shareSlug,
        sharedAt: now,
      };
    },
    { invalidates: ['notes'] }
  );

  return { shareNote: mutate, isLoading, error };
}
```

### 2. Unshare Note

```typescript
// modules/notes/api/mutations/unshare-note.ts
export function useUnshareNote() {
  const { mutate, isLoading, error } = useMutation(
    async (noteId: string) => {
      await transact([
        tx.notes[noteId].update({
          isPublic: false,
          shareSlug: null,
          sharedAt: null,
        }),
      ]);
    },
    { invalidates: ['notes'] }
  );

  return { unshareNote: mutate, isLoading, error };
}
```

### 3. Get Shared Note (Public)

```typescript
// modules/share/api/queries/get-shared-note.ts
export function useSharedNote(slug: string) {
  const { data, isLoading, error } = useRead({
    notes: {
      $: {
        where: {
          shareSlug: slug,
          isPublic: true,
        },
      },
    },
  });

  const note = data?.notes?.[0];
  
  return {
    note: note ? {
      title: note.title,
      content: note.content,
      sharedAt: note.sharedAt,
    } : null,
    isLoading,
    error: !note && !isLoading ? new Error('Note not found') : error,
  };
}
```

## Security Checklist

### What NOT to expose in shared view:
- ❌ User information (ID, email, name)
- ❌ Other notes
- ❌ Tasks (unless explicitly enabled)
- ❌ Internal database IDs
- ❌ Edit history
- ❌ Navigation to other notes

### What to expose:
- ✅ Note title
- ✅ Note content (sanitized)
- ✅ Share date

### Additional Security:
1. **Validate slug format** - Only alphanumeric, 10 chars
2. **Sanitize HTML** - Use DOMPurify on content
3. **Rate limit sharing** - Max 10 shares/hour per user
4. **No indexing** - Add `robots: noindex` meta tag

```typescript
// lib/sanitize.ts
import DOMPurify from 'dompurify';

export function sanitizeContent(html: string): string {
  return DOMPurify.sanitize(html, {
    ALLOWED_TAGS: ['p', 'br', 'strong', 'em', 'u', 'h1', 'h2', 'h3', 
                   'ul', 'ol', 'li', 'a', 'code', 'pre'],
    ALLOWED_ATTR: ['href', 'class'],
  });
}
```
